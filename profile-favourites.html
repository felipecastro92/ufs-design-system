FavouritesGrid components/profile/FavouritesGrid.tsx 
import React, { useState } from 'react';

type FavTab = 'recipes' | 'products' | 'training';

interface FavItem {
  id: string; type: FavTab; title: string; imageUrl: string; rating?: number;
}

interface FavouritesGridProps {
  items: FavItem[];
  onUnfavourite?: (id: string) => void;
}

export const FavouritesGrid: React.FC<FavouritesGridProps> = ({ items, onUnfavourite }) => {
  const [tab, setTab] = useState<FavTab>('recipes');
  const visible = items.filter(i => i.type === tab);

  return (
    <section className="favourites-section">
      <div className="rs-tabs">
        {(['recipes', 'products', 'training'] as FavTab[]).map(t => (
          <button
            key={t}
            className={`rs-tab${tab === t ? ' active' : ''}`}
            onClick={() => setTab(t)}
          >
            {t.charAt(0).toUpperCase() + t.slice(1)}
          </button>
        ))}
      </div>
      <div className="content-tiles">
        {visible.map(item => (
          <div key={item.id} className="ctile">
            <div className="ctile-img">
              <img src={item.imageUrl} alt={item.title} />
              <button className="ctile-fav" data-saved="true"
                onClick={() => onUnfavourite?.(item.id)} />
            </div>
            <div className="ctile-body"><div className="ctile-title">{item.title}</div></div>
          </div>
        ))}
      </div>
    </section>
  );
};
