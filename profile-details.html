import React, { useState } from 'react';

interface ProfileValues {
  firstName: string; lastName: string; email: string;
  phone: string; company: string; vatNumber: string;
}

interface ProfileDetailsFormProps {
  initialValues?: Partial<ProfileValues>;
  onSubmit?: (values: ProfileValues) => void;
}

export const ProfileDetailsForm: React.FC<ProfileDetailsFormProps> = ({ initialValues = {}, onSubmit }) => {
  const [values, setValues] = useState<ProfileValues>({
    firstName: '', lastName: '', email: '', phone: '', company: '', vatNumber: '',
    ...initialValues,
  });

  const handle = (e: React.ChangeEvent<HTMLInputElement>) =>
    setValues(prev => ({ ...prev, [e.target.name]: e.target.value }));

  return (
    <form className="pref-form" onSubmit={e => { e.preventDefault(); onSubmit?.(values); }}>
      {(['firstName', 'lastName', 'email', 'phone', 'company', 'vatNumber'] as (keyof ProfileValues)[]).map(field => (
        <div key={field} className="fi-wrap">
          <input className="fi-field" id={field} name={field}
            type="text" placeholder=" " value={values[field]} onChange={handle} />
          <label className="fi-label" htmlFor={field}>{field}</label>
        </div>
      ))}
      <button type="submit" className="btn btn-primary">Save changes</button>
    </form>
  );
};
