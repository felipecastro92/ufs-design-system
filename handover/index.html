Overview

Introduction

The UFS Design System is the single source of truth for all digital experiences created for Unilever Food Solutions. It provides designers, developers, and marketers with the tools and guidelines needed to build consistent, accessible, and on-brand digital products.

🎨
For Designers

A structured visual language with defined tokens for color, typography, spacing, and more. Use these foundations as the building blocks for every UI decision.

💻
For Developers

CSS custom properties and React / styled-components tokens ready to copy and paste. All values map directly from the Figma source files.

📣
For Marketers

Brand guidelines that ensure every campaign and communication reflects the right look, feel, and voice of Unilever Food Solutions across all channels.

AI Gradient

Reserved exclusively for AI-powered interface elements. Apply as a border, glow, or box-shadow to signal intelligent functionality — never as a fill on large surfaces.

--gradient-ai · linear-gradient(135deg, #FF5A00 → #D536B0)
Start — UFS Orange #FF5A00 --color-brand-primary
End — AI Violet #D536B0 --color-ai-accent
Usage guidance ✦ Borders — 1–2 px gradient border on AI input fields or response containers
✦ Glows — soft radial blur behind AI action buttons or chip badges
✦ Shadows — box-shadow with a coloured spread to elevate AI-generated cards
✦ Never use as a background fill on large surfaces or body text
CSS
React / SC
CSS Custom Properties

  --color-ai-accent: #D536B0;

Icons are delivered as inline SVG. Use stroke="currentColor" to inherit the parent's color, and set width/height to control size.

/* Sizing via CSS class */
.icon {
  width: 24px;
  height: 24px;
  display: inline-block;
  vertical-align: middle;
}

/* Inherit color from parent */
.icon svg {
  stroke: currentColor;
  fill: none;
}
// Inline SVG — no extra dependencies needed
function IconSearch({ size = 24, color = 'currentColor', ...props }) {

  );
}

import React, { useRef } from 'react';

type AvatarSize = 'sm' | 'md' | 'lg';

interface AvatarProps {
  size: AvatarSize;
  initials?: string;
  src?: string;
  alt?: string;
  editable?: boolean;
  points?: number;
  onPhotoChange?: (file: File) => void;
}

export const Avatar: React.FC<AvatarProps> = ({ size, initials, src, alt, editable, points, onPhotoChange }) => {
  const fileRef = useRef<HTMLInputElement>(null);
  return (
    <div className={`av av--${size}`}>
      <div className="av-circle">
        {src
          ? <img className="av-img" src={src} alt={alt} />
          : <span className="av-initials">{initials}</span>}
      </div>
      {editable && (
        <button className="av-edit" onClick={() => fileRef.current?.click()} aria-label="Change photo">
          {/* pencil SVG */}
          <input ref={fileRef} type="file" accept="image/*" hidden
            onChange={e => e.target.files?.[0] && onPhotoChange?.(e.target.files[0])} />
        </button>
      )}
      {points !== undefined && (
        <div className="av-badge">
          {/* coin SVG */}
          <span>{points.toLocaleString()}</span>
        </div>
      )}
    </div>
  );
};

interface AccordionItemData {
  id: string;
  question: string;
  answer: string;
}

interface AccordionProps {
  title: string;
  items: AccordionItemData[];
  visibleCount?: number; // default 4
}

export const Accordion: React.FC<AccordionProps> = ({ title, items, visibleCount = 4 }) => {
  const [openId, setOpenId] = useState<string | null>(items[0]?.id ?? null);
  const [expanded, setExpanded] = useState(false);
  const visible = expanded ? items : items.slice(0, visibleCount);

  return (
    <>
      <h2 className="ufs-accordion__title">{title}</h2>
      <div className={`ufs-accordion__panel${expanded ? ' is-expanded' : ''}`}>
        {visible.map(item => (
          <div key={item.id} className={`ufs-accordion__item${openId === item.id ? ' is-open' : ''}`}>
            <button className="ufs-accordion__trigger" aria-expanded={openId === item.id}
              onClick={() => setOpenId(openId === item.id ? null : item.id)}>
              <span className="ufs-accordion__q">{item.question}</span>
              {/* chevron svg */}
            </button>
            <div className="ufs-accordion__content">
              <div className="ufs-accordion__content-inner">
                <p className="ufs-accordion__a">{item.answer}</p>
              </div>
            </div>
          </div>
        ))}
        {items.length > visibleCount && (
          <button className="ufs-accordion__more" onClick={() => setExpanded(!expanded)}>
            {expanded ? 'Show less' : `Show all ${items.length} Frequently Asked Questions`}
          </button>
        )}
      </div>
    </>
  );
};
