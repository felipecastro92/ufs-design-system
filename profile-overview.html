ProfileTabsBar components/profile/ProfileTabsBar.tsx 
import React from 'react';

type ProfileTab = 'overview' | 'details' | 'orders' | 'favourites';

interface ProfileTabsBarProps { active: ProfileTab; }

const TABS: { id: ProfileTab; label: string; href: string }[] = [
  { id: 'overview',   label: 'Overview',      href: '/profile' },
  { id: 'details',    label: 'My profile',    href: '/profile/details' },
  { id: 'orders',     label: 'My Orders',     href: '/profile/orders' },
  { id: 'favourites', label: 'My Favourites', href: '/profile/favourites' },
];

export const ProfileTabsBar: React.FC<ProfileTabsBarProps> = ({ active }) => (
  <div className="tabs-bar">
    <div className="tabs-inner" role="tablist">
      {TABS.map(tab => (
        <a
          key={tab.id}
          href={tab.href}
          className={`tab-item${active === tab.id ? ' active' : ''}`}
          role="tab"
          aria-selected={active === tab.id}
        >{tab.label}</a>
      ))}
    </div>
  </div>
);

MyFavouritesCard components/profile/MyFavouritesCard.tsx 
import React, { useState } from 'react';

type FavTab = 'recipes' | 'products' | 'training';

interface MyFavouritesCardProps {
  stats: { recipes: number; products: number; training: number };
  recentItems: { id: string; type: FavTab; title: string; imageUrl: string }[];
}

export const MyFavouritesCard: React.FC<MyFavouritesCardProps> = ({ stats, recentItems }) => {
  const [tab, setTab] = useState<FavTab>('recipes');
  return (
    <div className="p-card">
      <a href="/profile/favourites" className="p-card-hdr">My Favourites</a>
      <div className="p-card-body">
        <div className="rs-tabs">
          {(['recipes', 'products', 'training'] as FavTab[]).map(t => (
            <button key={t} className={`rs-tab${tab === t ? ' active' : ''}`} onClick={() => setTab(t)}>
              {stats[t]} {t}
            </button>
          ))}
        </div>
        <div className="content-tiles">
          {recentItems.filter(i => i.type === tab).map(item => (
            <div key={item.id} className="ctile">
              <div className="ctile-img"><img src={item.imageUrl} alt={item.title} /></div>
              <div className="ctile-body"><div className="ctile-title">{item.title}</div></div>
            </div>
          ))}
        </div>
      </div>
    </div>
  );
};
