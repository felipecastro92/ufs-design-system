OrderCard components/profile/OrderCard.tsx 
import React from 'react';

interface OrderCardProps {
  id: string; date: string; itemCount: number;
  total: string; currency: string;
  thumbnails: { src: string; alt: string }[];
  extraCount?: number;
  onReorder?: () => void;
}

export const OrderCard: React.FC<OrderCardProps> = ({
  id, date, itemCount, total, currency, thumbnails, extraCount, onReorder,
}) => (
  <a href={`/profile/orders/${id}`} className="order-card">
    <div className="order-card-top">
      <div className="order-title-row">
        <div className="order-id">Order: {id}</div>
        <div className="order-date">{date}</div>
      </div>
      <div className="order-meta">{itemCount} items &middot; {currency} {total}</div>
    </div>
    <div className="order-thumbnails">
      {thumbnails.map((t, i) => (
        <div key={i} className="order-thumb"><img src={t.src} alt={t.alt} /></div>
      ))}
      {extraCount && <div className="order-thumb-more">+{extraCount}</div>}
    </div>
    <button
      className="btn btn-secondary btn-sm"
      onClick={e => { e.preventDefault(); onReorder?.(); }}
    >Reorder</button>
  </a>
);
